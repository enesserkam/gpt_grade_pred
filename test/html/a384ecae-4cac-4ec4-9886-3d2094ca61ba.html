To find the information gain on the first split with entropy, you need to calculate the entropy of the parent node and the average entropy of the child nodes. The formula for information gain is:

Here's how you can calculate it:

from sklearn.tree import _tree

def calculate_entropy(y):
    _, counts = np.unique(y, return_counts=True)
    probabilities = counts / len(y)
    entropy = -np.sum(probabilities * np.log2(probabilities))
    return entropy

def calculate_information_gain(y, y_left, y_right):
    entropy_parent = calculate_entropy(y)
    entropy_left = calculate_entropy(y_left)
    entropy_right = calculate_entropy(y_right)

    # Calculate the average entropy of the children
    average_entropy_children = (len(y_left) / len(y)) * entropy_left + (len(y_right) / len(y)) * entropy_right

    # Calculate information gain
    information_gain = entropy_parent - average_entropy_children
    return information_gain
